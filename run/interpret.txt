object Math {
    define init(inst, num) {
        inst.num = num
    }
    define factorial(inst, number) {
        original = inst.num
        inst.num = inst.num - 1
        if original == 1 {
            return 1
        }
        else {
            return original * inst.factorial(inst.num)
        }
    }
}

foo = Math(15)
a = foo.factorial()
a
